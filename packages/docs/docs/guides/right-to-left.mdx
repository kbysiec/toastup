## Example

```tsx
import { toast, Toaster } from "@toastup/react";
import "@toastup/react/style";

export function RtlExample() {
  const handleClick = () => toast.add({ rtl: true });
  return (
    <div>
      <button onClick={handleClick}>Add toast</button>
      <Toaster />
    </div>
  );
}
```
