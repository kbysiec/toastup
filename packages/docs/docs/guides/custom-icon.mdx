Allows to render a custom component as an icon.

import { CustomIconExample } from "@site/src/components/examples/CustomIconExample";

<CustomIconExample />

## Example

```tsx
import CustomIcon from "@site/static/img/toast.svg";
import { toast, Toaster } from "@toastup/react";

export function CustomIconExample() {
  const handleClick = () =>
    toast.add({
      icon: () => <CustomIcon className={styles.customIcon} />,
    });
  return (
    <div>
      <button onClick={handleClick}>Add toast</button>
      <Toaster />
    </div>
  );
}
```
